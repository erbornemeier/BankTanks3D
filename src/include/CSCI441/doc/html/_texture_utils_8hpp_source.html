145 inline bool CSCI441::TextureUtils::loadBMP( const char* filename, int &imageWidth, int &imageHeight, int &imageChannels, unsigned char* imageData, const char* path ) {
146  FILE *file;
147  unsigned long size; // size of the image in bytes.
148  size_t i; // standard counter.
149  unsigned short int planes; // number of planes in image (must be 1)
150  unsigned short int bpp; // number of bits per pixel (must be 24)
151  char temp; // used to convert bgr to rgb color.
152 
153  // make sure the file is there.
154  if ((file = fopen(filename, "rb"))==NULL) {
155  string folderName = string(path) + string(filename);
156  if ((file = fopen(folderName.c_str(), "rb")) == NULL ) {
157  printf("[.bmp]: [ERROR]: File Not Found: %s\n",filename);
158  return false;
159  }
160  }
161 
162  // seek through the bmp header, up to the width/height:
163  fseek(file, 18, SEEK_CUR);
164 
165  // read the width
166  if ((i = fread(&imageWidth, 4, 1, file)) != 1) {
167  printf("[.bmp]: [ERROR]: reading width from %s.\n", filename);
168  return false;
169  }
170  //printf("Width of %s: %lu\n", filename, image->sizeX);
171 
172  // read the height
173  if ((i = fread(&imageHeight, 4, 1, file)) != 1) {
174  printf("[.bmp]: [ERROR]: reading height from %s.\n", filename);
175  return false;
176  }
177  //printf("Height of %s: %lu\n", filename, image->sizeY);
178 
179  // calculate the size (assuming 24 bits or 3 bytes per pixel).
180  size = imageWidth * imageHeight * 3;
181 
182  // read the planes
183  if ((fread(&planes, 2, 1, file)) != 1) {
184  printf("[.bmp]: [ERROR]: reading planes from %s.\n", filename);
185  return false;
186  }
187  if (planes != 1) {
188  printf("[.bmp]: [ERROR]: Planes from %s is not 1: %u\n", filename, planes);
189  return false;
190  }
191 
192  // read the bpp
193  if ((i = fread(&bpp, 2, 1, file)) != 1) {
194  printf("[.bmp]: [ERROR]: reading bpp from %s.\n", filename);
195  return false;
196  }
197  if (bpp != 24) {
198  printf("[.bmp]: [ERROR]: Bpp from %s is not 24: %u\n", filename, bpp);
199  return false;
200  }
201 
202  // seek past the rest of the bitmap header.
203  fseek(file, 24, SEEK_CUR);
204 
205  // read the data.
206  imageData = (unsigned char *) malloc(size);
207  if (imageData == NULL) {
208  printf("[.bmp]: [ERROR]: allocating memory for color-corrected image data");
209  return false;
210  }
211 
212  if ((i = fread(imageData, size, 1, file)) != 1) {
213  printf("[.bmp]: [ERROR]: reading image data from %s.\n", filename);
214  return false;
215  }
216 
217  for (i=0;i<size;i+=3) { // reverse all of the colors. (bgr -> rgb)
218  temp = imageData[i];
219  imageData[i] = imageData[i+2];
220  imageData[i+2] = temp;
221  }
222 
223  imageChannels = 3;
224 
225  return true;

228 inline bool CSCI441::TextureUtils::loadPPM( const char *filename, int &imageWidth, int &imageHeight, unsigned char* &imageData ) {
229  FILE *fp = fopen(filename, "r");
230  int temp, maxValue;
231  fscanf(fp, "P%d", &temp);
232  if(temp != 3) {
233  fprintf(stderr, "Error: PPM file is not of correct format! (Must be P3, is P%d.)\n", temp);
234  fclose(fp);
235  return false;
236  }
237 
238  //got the file header right...
239  fscanf(fp, "%d", &imageWidth);
240  fscanf(fp, "%d", &imageHeight);
241  fscanf(fp, "%d", &maxValue);
242 
243  //now that we know how big it is, allocate the buffer...
244  imageData = new unsigned char[imageWidth*imageHeight*3];
245  if(!imageData) {
246  fprintf(stderr, "Error: couldn't allocate image memory. Dimensions: %d x %d.\n", imageWidth, imageHeight);
247  fclose(fp);
248  return false;
249  }
250 
251  //and read the data in.
252  for(int j = 0; j < imageHeight; j++) {
253  for(int i = 0; i < imageWidth; i++) {
254  int r, g, b;
255  fscanf(fp, "%d", &r);
256  fscanf(fp, "%d", &g);
257  fscanf(fp, "%d", &b);
258 
259  imageData[(j*imageWidth+i)*3+0] = r;
260  imageData[(j*imageWidth+i)*3+1] = g;
261  imageData[(j*imageWidth+i)*3+2] = b;
262  }
263  }
264 
265  fclose(fp);
266  return true;

269 inline bool CSCI441::TextureUtils::loadTGA(const char *filename, int &imageWidth, int &imageHeight, unsigned char* &imageData, int &imageChannels ) {
270  FILE *fp = fopen(filename, "rb");
271  if(!fp) {
272  fprintf(stderr, "Error: could not open TGA file: %s.", filename);
273  imageData = NULL;
274  return false;
275  }
276 
277  //bunch of data fields in the file header that we read in en masse
278  unsigned char idLength, colorMapType, imageType;
279  unsigned short idxOfFirstColorMapEntry, countOfColorMapEntries;
280  unsigned char numBitsPerColorMapEntry;
281  unsigned short xCoordOfLowerLeft, yCoordOfLowerLeft;
282  unsigned short width, height;
283  unsigned char bytesPerPixel;
284  unsigned char imageAttributeFlags;
285 
286  fread(&idLength, sizeof(unsigned char), 1, fp);
287  fread(&colorMapType, sizeof(unsigned char), 1, fp);
288  fread(&imageType, sizeof(unsigned char), 1, fp);
289  fread(&idxOfFirstColorMapEntry, sizeof(unsigned short), 1, fp);
290  fread(&countOfColorMapEntries, sizeof(unsigned short), 1, fp);
291  fread(&numBitsPerColorMapEntry, sizeof(unsigned char), 1, fp);
292  fread(&xCoordOfLowerLeft, sizeof(unsigned short), 1, fp);
293  fread(&yCoordOfLowerLeft, sizeof(unsigned short), 1, fp);
294  fread(&width, sizeof(unsigned short), 1, fp);
295  fread(&height, sizeof(unsigned short), 1, fp);
296  fread(&bytesPerPixel, sizeof(unsigned char), 1, fp);
297  fread(&imageAttributeFlags, sizeof(unsigned char), 1, fp);
298 
299  //now check to make sure that we actually have the capability to read this file.
300  if(colorMapType != 0) {
301  fprintf(stderr, "Error: TGA file (%s) uses colormap instead of RGB/RGBA data; this is unsupported.\n", filename);
302  imageData = NULL;
303  return false;
304  }
305 
306  if(imageType != 2 && imageType != 10) {
307  fprintf(stderr, "Error: unspecified TGA type: %d. Only supports 2 (uncompressed RGB/A) and 10 (RLE, RGB/A).\n", imageType);
308  imageData = NULL;
309  return false;
310  }
311 
312  if(bytesPerPixel != 24 && bytesPerPixel != 32) {
313  fprintf(stderr, "Error: unsupported image depth (%d bits per pixel). Only supports 24bpp and 32bpp.\n", bytesPerPixel);
314  imageData = NULL;
315  return false;
316  }
317 
318 
319  //set some helpful variables based on the header information:
320  bool usingRLE = (imageType == 10); //whether the file uses run-length encoding (compression)
321  imageChannels = (bytesPerPixel == 32 ? 4 : 3); //whether the file is RGB or RGBA (RGBA = 32bpp)
322  bool topLeft = (imageAttributeFlags & 32); //whether the origin is at the top-left or bottom-left
323 
324 
325  //this should never happen, since we should never have a color map,
326  //but just in case the data is setting around in there anyway... skip it.
327  if(idLength != 0) {
328  fseek(fp, idLength, SEEK_CUR);
329  }
330 
331 
332  //allocate our image data before we get started.
333  imageWidth = width;
334  imageHeight = height;
335  imageData = new unsigned char[imageWidth*imageHeight*imageChannels];
336 
337  //ok so we can assume at this point that there's no colormap, and
338  //consequently that the next part of the image is the actual image data.
339  if(usingRLE) {
340  //ok... the data comes in in packets, but we don't know how many of these there'll be.
341  int numberOfPixelsRead = 0;
342  unsigned char dummy;
343  while(numberOfPixelsRead < imageWidth*imageHeight) {
344  //alright let's read the packet header.
345  fread(&dummy, sizeof(unsigned char), 1, fp);
346  bool isRLEPacket = (dummy & 0x80);
347 
348  unsigned char theOtherBitsYesThatWasAPun = (dummy & 0x7F);
349  if(isRLEPacket) {
350  //the other bits (+1) gives the number of times we need to
351  //repeat the next real set of color values (RGB/A).
352  unsigned char repeatedR, repeatedG, repeatedB, repeatedA;
353  fread(&repeatedR, sizeof(unsigned char), 1, fp);
354  fread(&repeatedG, sizeof(unsigned char), 1, fp);
355  fread(&repeatedB, sizeof(unsigned char), 1, fp);
356  if(imageChannels==4) fread(&repeatedA, sizeof(unsigned char), 1, fp);
357 
358  //and go ahead and copy those into the new image. repetitively.
359  for(int i = 0; i < ((int)theOtherBitsYesThatWasAPun+1); i++) {
360  int idx = numberOfPixelsRead * imageChannels;
361  imageData[idx+2] = repeatedR;
362  imageData[idx+1] = repeatedG;
363  imageData[idx+0] = repeatedB;
364  if(imageChannels==4) imageData[idx+3] = repeatedA;
365 
366  numberOfPixelsRead++;
367  }
368  } else {
369  //the other bits (+1) gives the number of consecutive
370  //pixels we get to read in from the stream willy nilly.
371  for(int i = 0; i < ((int)theOtherBitsYesThatWasAPun+1); i++) {
372  int idx = numberOfPixelsRead * imageChannels;
373  fread(&imageData[idx+2], sizeof(unsigned char), 1, fp);
374  fread(&imageData[idx+1], sizeof(unsigned char), 1, fp);
375  fread(&imageData[idx+0], sizeof(unsigned char), 1, fp);
376  if(imageChannels==4) fread(&imageData[idx+3], sizeof(unsigned char), 1, fp);
377 
378  numberOfPixelsRead++;
379  }
380  }
381  }
382 
383 
384  //and you know what? we're not going to have worried about flipping the image before
385  //if its origin was in the bottom left or top left or whatever. flip it afterwards here if need be.
386  if(!topLeft) {
387  unsigned char *tempCopy = new unsigned char[imageWidth*imageHeight*imageChannels];
388  for(int i = 0; i < imageHeight; i++) {
389  for(int j = 0; j < imageWidth; j++) {
390  int copyIdx = (i*imageWidth + j)*imageChannels;
391  int pullIdx = ((imageHeight - i - 1)*imageWidth + j)*imageChannels;
392  tempCopy[copyIdx+0] = imageData[pullIdx+0];
393  tempCopy[copyIdx+1] = imageData[pullIdx+1];
394  tempCopy[copyIdx+2] = imageData[pullIdx+2];
395  if(imageChannels==4) tempCopy[copyIdx+3] = imageData[pullIdx+3];
396  }
397  }
398 
399  delete imageData;
400  imageData = tempCopy;
401  }
402 
403  } else {
404  //uh well if we're not using run-length encoding, i guess we'll
405  //just try reading bytes in straight like a normal binary file.
406  unsigned char byte1, byte2, byte3, maybeEvenByte4;
407  for(int i = 0; i < imageHeight; i++) {
408  for(int j = 0; j < imageWidth; j++) {
409  int multiplierThing = imageChannels;
410 
411  //read in the data from file...
412  fread(&byte1, sizeof(unsigned char), 1, fp);
413  fread(&byte2, sizeof(unsigned char), 1, fp);
414  fread(&byte3, sizeof(unsigned char), 1, fp);
415  if(imageChannels==4) fread(&maybeEvenByte4, sizeof(unsigned char), 1, fp);
416 
417  //flip the vertical index if the origin is in the bottom-left.
418  int wutHeight = topLeft ? i : (imageHeight - 1 - i);
419  int idx = (wutHeight*imageWidth+j)*multiplierThing;
420 
421  //and load that image into file. seems to be BGR instead of RGB...
422  imageData[idx+2] = byte1;
423  imageData[idx+1] = byte2;
424  imageData[idx+0] = byte3;
425  if(imageChannels==4) imageData[idx+3] = maybeEvenByte4;
426  }
427  }
428  }
429 
430  fclose(fp);
431  return true;
432 }

448 inline GLuint CSCI441::TextureUtils::loadAndRegister2DTexture( const char *filename, GLenum minFilter, GLenum magFilter, GLenum wrapS, GLenum wrapT ) {
449  GLuint texHandle = SOIL_load_OGL_texture( filename,
450  SOIL_LOAD_AUTO,
451  SOIL_CREATE_NEW_ID,
452  SOIL_FLAG_MIPMAPS | SOIL_FLAG_INVERT_Y | SOIL_FLAG_NTSC_SAFE_RGB | SOIL_FLAG_COMPRESS_TO_DXT );
453 
454  if( texHandle == 0 ) {
455  printf( "[ERROR]: Could not load texture \"%s\"\n[SOIL]: %s\n", filename, SOIL_last_result() );
456  } else {
457  printf( "[INFO]: Successfully loaded texture \"%s\"\n[SOIL]: %s\n", filename, SOIL_last_result() );
458  glBindTexture( GL_TEXTURE_2D, texHandle );
459  glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, minFilter );
460  glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, magFilter );
461  glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, wrapS );
462  glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, wrapT );
463  }
464 
465  return texHandle;
466 }
