232  /* compile each one of our shaders */

258  }

272  if( sDEBUG ) printf( "[INFO]: | Fragment Shader: %37s |\n", fragmentShaderFilename );

275  /* get a handle to a shader program */

290 
291  /* link all the programs together on the GPU */
292  glLinkProgram( _shaderProgramHandle );
293 
294  if( sDEBUG ) printf( "[INFO]: | Shader Program: %41s", "|\n" );
295
