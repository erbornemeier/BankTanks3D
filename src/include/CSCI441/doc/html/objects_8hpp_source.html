486 inline void CSCI441_INTERNAL::drawCylinder( GLdouble base, GLdouble top, GLdouble height, GLint stacks, GLint slices, GLenum renderMode ) {
487  double sliceStep = 2.0 * M_PI / slices;
488  double stackStep = height / stacks;
489 
490  glPushAttrib( GL_POLYGON_BIT ); {
491 
492  glPolygonMode( GL_FRONT_AND_BACK, renderMode );
493 
494  for( int stackNum = 0; stackNum < stacks; stackNum++ ) {
495  GLdouble botRadius = base*(stacks-stackNum)/stacks + top*stackNum/stacks;
496  GLdouble topRadius = base*(stacks-stackNum-1)/stacks + top*(stackNum+1)/stacks;
497 
498  glBegin( GL_TRIANGLE_STRIP ); {
499  for( int sliceNum = 0; sliceNum <= slices; sliceNum++ ) {
500  glNormal3f( cos( sliceNum * sliceStep ), 0.0f, sin( sliceNum * sliceStep ) );
501 
502  glTexCoord2f( (double)sliceNum / slices, (double)stackNum / stacks );
503  glVertex3f( cos( sliceNum * sliceStep )*botRadius, stackNum * stackStep, sin( sliceNum * sliceStep )*botRadius );
504 
505  glTexCoord2f( (double)sliceNum / slices, (double)(stackNum+1) / stacks );
506  glVertex3f( cos( sliceNum * sliceStep )*topRadius, (stackNum+1) * stackStep, sin( sliceNum * sliceStep )*topRadius );
507  }
508  }; glEnd();
509  }
510 
511  }; glPopAttrib();

514 inline void CSCI441_INTERNAL::drawPartialDisk( GLdouble inner, GLdouble outer, GLint slices, GLint rings, GLdouble start, GLdouble sweep, GLenum renderMode ) {
515  double sliceStep = sweep / slices;
516  double ringStep = (outer - inner) / rings;
517 
518  glPushAttrib( GL_POLYGON_BIT ); {
519 
520  glPolygonMode( GL_FRONT_AND_BACK, renderMode );
521 
522  for( int ringNum = 0; ringNum < rings; ringNum++ ) {
523  double currRadius = inner + ringNum*ringStep;
524  double nextRadius = inner + (ringNum+1)*ringStep;
525 
526  glBegin( GL_TRIANGLE_STRIP ); {
527  glNormal3f( 0.0f, 0.0f, 1.0f );
528 
529  double theta = start;
530  for( int i = 0; i <= slices; i++, theta += sliceStep ) {
531  glTexCoord2f( cos(theta)*(currRadius/outer), sin(theta)*(currRadius/outer) );
532  glVertex3f( cos(theta)*currRadius, sin(theta)*currRadius, 0.0f );
533 
534  glTexCoord2f( cos(theta)*(nextRadius/outer), sin(theta)*(nextRadius/outer) );
535  glVertex3f( cos(theta)*nextRadius, sin(theta)*nextRadius, 0.0f );
536  }
537  }; glEnd();
538  }
539 
540  }; glPopAttrib();

543 inline void CSCI441_INTERNAL::drawSphere( GLdouble radius, GLint stacks, GLint slices, GLenum renderMode ) {
544  double sliceStep = 2.0 * M_PI / slices;
545  double stackStep = M_PI / stacks;
546 
547  glPushAttrib( GL_POLYGON_BIT ); {
548 
549  glPolygonMode( GL_FRONT_AND_BACK, renderMode );
550 
551  // sphere top
552  glBegin( GL_TRIANGLE_FAN ); {
553  double phi = stackStep * stacks;
554  double phiNext = stackStep * (stacks-1);
555 
556  glNormal3f( 0.0f, 1.0f, 0.0f );
557  glTexCoord2f( 0.5f, 1.0f );
558  glVertex3f( 0.0f, -cos( phi )*radius, 0.0f );
559 
560  for( int sliceNum = 0; sliceNum <= slices; sliceNum++ ) {
561  double theta = sliceStep * sliceNum;
562 
563  glNormal3f( -cos( theta )*sin( phiNext ), -cos( phiNext ), sin( theta )*sin( phiNext ) );
564  glTexCoord2f( (stacks-1.0f) / stacks, sliceNum / slices );
565  glVertex3f( -cos( theta )*sin( phiNext )*radius, -cos( phiNext )*radius, sin( theta )*sin( phiNext )*radius );
566  }
567  }; glEnd();
568 
569  // sphere stacks
570  for( int stackNum = 1; stackNum < stacks-1; stackNum++ ) {
571  double phi = stackStep * stackNum;
572  double phiNext = stackStep * (stackNum + 1);
573 
574  glBegin( GL_TRIANGLE_STRIP ); {
575  for( int sliceNum = slices; sliceNum >= 0; sliceNum-- ) {
576  double theta = sliceStep * sliceNum;
577 
578  glNormal3f( -cos( theta )*sin( phi ), -cos( phi ), sin( theta )*sin( phi ) );
579  glTexCoord2f( stackNum / sphereData.st, sliceNum / sphereData.sl );
580  glVertex3f( -cos( theta )*sin( phi )*radius, -cos( phi )*radius, sin( theta )*sin( phi )*radius );
581 
582  glNormal3f( -cos( theta )*sin( phiNext ), -cos( phiNext ), sin( theta )*sin( phiNext ) );
583  glTexCoord2f( (stackNum+1) / sphereData.st, sliceNum / sphereData.sl );
584  glVertex3f( -cos( theta )*sin( phiNext )*radius, -cos( phiNext )*radius, sin( theta )*sin( phiNext )*radius );
585  }
586  }; glEnd();
587  }
588 
589  // sphere bottom
590  glBegin( GL_TRIANGLE_FAN ); {
591  double phi = 0;
592  double phiNext = stackStep;
593 
594  glNormal3f( 0.0f, -1.0f, 0.0f );
595  glTexCoord2f( 0.5f, 0.0f );
596  glVertex3f( 0.0f, -cos( phi )*radius, 0.0f );
597 
598  for( int sliceNum = slices; sliceNum >= 0; sliceNum-- ) {
599  double theta = sliceStep * sliceNum;
600 
601  glNormal3f( -cos( theta )*sin( phiNext ), -cos( phiNext ), sin( theta )*sin( phiNext ) );
602  glTexCoord2f( 1.0f / stacks, sliceNum / slices );
603  glVertex3f( -cos( theta )*sin( phiNext )*radius, -cos( phiNext )*radius, sin( theta )*sin( phiNext )*radius );
604  }
605  }; glEnd();
606 
607  }; glPopAttrib();
608 }
