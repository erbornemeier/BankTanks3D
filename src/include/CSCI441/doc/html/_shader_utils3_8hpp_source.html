383 GLuint CSCI441_INTERNAL::ShaderUtils::compileShader( const char *filename, GLenum shaderType ) {
384  GLuint shaderHandle = 0;
385  char *shaderString;
386 
387  /* create a handle to our shader */
388  shaderHandle = glCreateShader( shaderType );
389 
390  /* read in each text file and store the contents in a string */
391  readTextFromFile( filename, shaderString );
392 
393  /* send the contents of each program to the GPU */
394  glShaderSource( shaderHandle, 1, (const char**)&shaderString, NULL );
395 
396  /* we are good programmers so free up the memory used by each buffer */
397  delete [] shaderString;
398 
399  /* compile each shader on the GPU */
400  glCompileShader( shaderHandle );
401 
402  /* check the shader log */
403  printLog( shaderHandle );
404 
405  /* return the handle of our shader */
406  return shaderHandle;
407 }
